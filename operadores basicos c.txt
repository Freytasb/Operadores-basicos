#include <stdio.h>

int main()
{
    int soma,divi,multi,sub,n1,n2;
    printf("Digite o primeiro número");
    scanf("%d",&n1);
    printf("Digite o segundo número");
    scanf("%d",&n2);
    soma=n1+n2;
    
    if  (n1<=n2)
    {
        sub=n2-n1;
    }
    else
    {
        sub=n1-n2;
    }
    if  (n1<=n2)
    {
        divi=n2/n1;
    }
    else
    {
        divi=n1/n2;
    }    
    multi=n1*n2;
    printf("O resultado da soma é: %d",soma);
    printf("O resultado da divisão é: %d",divi);
    printf("O resultado da multiplicação é: %d",multi);
    printf("O resultado da subtração é: %d",sub);
    return 0; 
}
